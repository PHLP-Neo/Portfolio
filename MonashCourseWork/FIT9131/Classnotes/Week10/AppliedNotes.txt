Null pointer exception
Index out of bound exception
Input mismatch exception

import java.util.Scanner;
import java.io.FileWriter;

public class WriteFile
{
    public static void main(String[] args)
    {
        boolean flag = true;
        Scanner console = new Scanner(System.in);
       
        try
        {
            FileWriter writer = new FileWriter("temp.txt");
            do
            {
                System.out.println("Enter text to append to file...");
                String value = console.nextLine();
                value += "\n";
                writer.append(value);
                try
                {
                    System.out.println("Continue adding text to file (y or n)?");
                    char response = console.nextLine().trim().toLowerCase().charAt(0);
                    if (response == 'n')
                    {
                        flag = false;
                        writer.close();
                    }
                }
                catch (Exception e)
                {
                    System.out.println("Please enter y or n only!");
                }
            } while (flag);
        }
        catch (Exception e)
        {
            System.out.println("Error in writing to file! Exiting...");
        }
    }
}
